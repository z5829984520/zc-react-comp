## Spinner

### Line

```tsx
import React from 'react';
import { Spinner } from 'zc-react-comp';

export default () => <Spinner />;
```

### Point

```tsx
import React from 'react';
import { Spinner } from 'zc-react-comp';

export default () => <Spinner type="point" />;
```

### Text

```tsx
import React from 'react';
import { Spinner } from 'zc-react-comp';

export default () => <Spinner>Loading</Spinner>;
```
